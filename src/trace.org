#+begin_src vue
  <template>
    <div class="shining">
      <h1> Vue </h1>
    </div>
  </template>

  <script lang="ts">
   export default {
     data() {
       return {
         opacity: 0
       }
     },
   
     mounted() {
       setInterval(() => {
         this.opacity >= 1 && (this.opacity = 0)
         this.opacity += 0.2
       }, 300)
     }
   }
  </script>

  <style scoped>
   h1 {
     color: black;
     opacity: v-bind(opacity);
   }
  </style>

#+end_src
